# style attribute for styling HTML elements: Inline CSS:
        <tagname style="property:value;">

# Attributes: Every element will have some properties. Those are called attributes. 
        Eg: src, alt for <img> element, href, target for <a> element

    1. background-color :  for background color -> <body style="background-color:powderblue;">
    2. color : for text colors -> <p style="color:red;">This is a paragraph.</p>
    3. font-family : for text fonts -> <p style="font-family:courier;">This is a paragraph.</p>
    4. font-size : for text sizes -> <p style="font-size:160%;">This is a paragraph.</p>
    5. text-align : for text alignment -> <h1 style="text-align:center;">Centered Heading</h1>
